void createHorizontalLayout(ComponentStorage& storage, LayoutEngine& engine) {

    auto* btnEntry = storage.getComponent(button);
    btnEntry->layoutResult = {100, 100, 120, 40};

    if (clicked.isValid()) {

        if (clickedEntry && clickedEntry->enabled) {
            printf("Button clicked!\n");
        }
    }
}

### 2. 鼠标悬停

**文件**: `examples/events/mouse_hover.cpp`

```cpp
#include "aether/AetherApplication.h"

void handleMouseHover(ComponentStorage& storage, EventDispatcher& dispatcher) {
    ComponentHandle lastHovered = INVALID_COMPONENT_HANDLE;
    
    // 模拟鼠标移动
    Point mousePos = GetMousePosition();
    auto hovered = dispatcher.hitTest(mousePos);
    
    if (hovered.isValid()) {
        if (!lastHovered.isValid() || lastHovered != hovered) {
            // 鼠标进入
            printf("Mouse entered component %d\n", hovered.index);
            lastHovered = hovered;
        }
    } else if (lastHovered.isValid()) {
        // 鼠标离开
        printf("Mouse left component %d\n", lastHovered.index);
        lastHovered = INVALID_COMPONENT_HANDLE;
    }
}
```

### 3. 键盘导航

**文件**: `examples/events/keyboard_nav.cpp`

```cpp
#include "aether/AetherApplication.h"

void handleKeyboard(ComponentStorage& storage, EventDispatcher& dispatcher) {
    // 创建多个可聚焦组件
    std::vector<ComponentHandle> focusableComponents;
    
    for (int i = 0; i < 5; i++) {
        auto button = storage.createComponent(ComponentType::Button);
        auto* btnEntry = storage.getComponent(button);
        btnEntry->focused = false; // 默认不聚焦
        focusableComponents.push_back(button);
    }
    
    // Tab键切换焦点
    void OnKeyDown(int keyCode) {
        if (keyCode == VK_TAB) {
            auto currentFocus = dispatcher.getFocusedComponent();
            
            if (!currentFocus.isValid()) {
                // 第一个获得焦点
                dispatcher.setFocusedComponent(focusableComponents[0]);
            } else {
                // 找到当前焦点索引
                for (size_t i = 0; i < focusableComponents.size(); i++) {
                    if (focusableComponents[i] == currentFocus) {
                        // 移动到下一个
                        size_t next = (i + 1) % focusableComponents.size();
                        dispatcher.setFocusedComponent(focusableComponents[next]);
                        break;
                    }
                }
            }
        }
    }
}
```

#include "aether/Direct2DRenderer.h"

void renderButton(Direct2DRenderer& renderer, ComponentEntry* entry) {
    renderer.beginDraw();
    
    // 清除画布
    renderer.clear(Color::White());
    
    // 绘制按钮背景
    Rect rect = entry->layoutResult;
    Color bgColor = Color::fromBytes(0, 120, 215); // Windows蓝
    renderer.fillRoundedRect(rect, 4.0f, bgColor);
    
    // 绘制文本
    if (auto* text = entry->properties.getProperty(PropertyId::Text)) {
        if (text->is<std::string>()) {
            std::wstring wtext = StringToWString(text->get<std::string>());
            renderer.drawText(wtext, rect, Color::White(), 14.0f);
        }
    }
    
    renderer.endDraw();
}

            renderer.drawText(

                theme.textColor, 
                theme.fontSize

### 3. 列表渲染

**文件**: `examples/rendering/list_rendering.cpp`

```cpp
#include "aether/Direct2DRenderer.h"

void renderList(Direct2DRenderer& renderer, 
                ComponentStorage& storage, 
                ComponentHandle listContainer) {
    renderer.beginDraw();
    renderer.clear(Color::White());
    
    auto* container = storage.getComponent(listContainer);
    
    float yOffset = 0;
    float itemHeight = 50;
    
    for (int32_t childIdx : container->childrenIndices) {
        auto* item = storage.getComponentByIndex(childIdx);
        if (!item || !item->visible) continue;
        
        // 绘制交替背景
        bool isEven = (yOffset / itemHeight) % 2 == 0;
        Color bgColor = isEven ? Color::White() : Color::fromBytes(240, 240, 240);
        
        Rect itemRect = {
            item->layoutResult.x,
            yOffset,
            item->layoutResult.width,
            itemHeight
        };
        
        renderer.fillRect(itemRect, bgColor);
        
        // 绘制分隔线
        renderer.drawRect(
            {itemRect.x, yOffset + itemHeight - 1, itemRect.width, 1},
            Color::fromBytes(200, 200, 200),
            1.0f
        );
        
        // 绘制文本
        if (auto* text = item->properties.getProperty(PropertyId::Text)) {
            if (text->is<std::string>()) {
                std::wstring wtext = StringToWString(text->get<std::string>());
                Rect textRect = {
                    itemRect.x + 16,
                    yOffset + 15,
                    itemRect.width - 32,
                    itemHeight - 30
                };
                renderer.drawText(wtext, textRect, Color::Black(), 14.0f);
            }
        }
        
        yOffset += itemHeight;
    }
    
    renderer.endDraw();
}
```

### 1. 待办事项应用

**文件**: `examples/apps/todo_app.cpp`

```cpp
#include "aether/AetherApplication.h"
#include <vector>
#include <string>

class TodoApp {
private:
    ComponentStorage& storage;
    LayoutEngine& engine;
    EventDispatcher& dispatcher;
    
    ComponentHandle rootContainer;
    ComponentHandle inputField;
    ComponentHandle addButton;
    ComponentHandle todoList;
    ComponentHandle footer;
    
    std::vector<ComponentHandle> todoItems;
    std::vector<std::string> todoTexts;

public:
    TodoApp(ComponentStorage& s, LayoutEngine& l, EventDispatcher& d)
        : storage(s), engine(l), dispatcher(d) {}
    
    void createUI() {
        // 创建根容器
        rootContainer = storage.createComponent(ComponentType::Container);
        auto* root = storage.getComponent(rootContainer);
        root->layoutResult = {0, 0, 400, 600};
        root->properties.setProperty(PropertyId::FlexDirection, 
                                    PropertyValue(FlexDirection::Column));
        
        // Header
        auto header = storage.createComponent(ComponentType::Container, rootContainer);
        auto* headerEntry = storage.getComponent(header);
        headerEntry->properties.setProperty(PropertyId::Height, PropertyValue(60));
        headerEntry->properties.setProperty(PropertyId::BackgroundColor, 
                                          PropertyValue(Color{0.3f, 0.6f, 0.9f, 1.0f}));
        
        auto title = storage.createComponent(ComponentType::Label, header);
        storage.getComponent(title)->properties.setProperty(
            PropertyId::Text, PropertyValue("待办事项"));
        
        // 输入区域
        auto inputContainer = storage.createComponent(
            ComponentType::Container, rootContainer);
        auto* inputEntry = storage.getComponent(inputContainer);
        inputEntry->properties.setProperty(PropertyId::FlexDirection, 
                                         PropertyValue(FlexDirection::Row));
        inputEntry->properties.setProperty(PropertyId::Padding, 
                                         PropertyValue(10));
        
        inputField = storage.createComponent(ComponentType::TextField, inputContainer);
        auto* inputFieldEntry = storage.getComponent(inputField);
        inputFieldEntry->properties.setProperty(PropertyId::FlexGrow, PropertyValue(1));
        inputFieldEntry->properties.setProperty(PropertyId::MarginRight, PropertyValue(10));
        inputFieldEntry->properties.setProperty(PropertyId::PlaceholderText, 
                                              PropertyValue("添加新待办..."));
        
        addButton = storage.createComponent(ComponentType::Button, inputContainer);
        storage.getComponent(addButton)->properties.setProperty(
            PropertyId::Width, PropertyValue(80));
        storage.getComponent(addButton)->properties.setProperty(
            PropertyId::Text, PropertyValue("添加"));
        
        // 待办列表
        todoList = storage.createComponent(ComponentType::Container, rootContainer);
        auto* todoListEntry = storage.getComponent(todoList);
        todoListEntry->properties.setProperty(PropertyId::FlexGrow, PropertyValue(1));
        todoListEntry->properties.setProperty(PropertyId::Padding, PropertyValue(10));
        
        // Footer
        footer = storage.createComponent(ComponentType::Container, rootContainer);
        auto* footerEntry = storage.getComponent(footer);
        footerEntry->properties.setProperty(PropertyId::Height, PropertyValue(40));
        
        std::string count = "共 " + std::to_string(todoItems.size()) + " 项";
        storage.getComponent(footer)->properties.setProperty(
            PropertyId::Text, PropertyValue(count));
        
        engine.forceRelayout();
        dispatcher.onLayoutComplete();
    }
    
    void addTodo(const std::string& text) {
        if (text.empty()) return;
        
        auto todo = storage.createComponent(ComponentType::Container, todoList);
        auto* todoEntry = storage.getComponent(todo);
        
        todoEntry->properties.setProperty(PropertyId::FlexDirection, 
                                         PropertyValue(FlexDirection::Row));
        todoEntry->properties.setProperty(PropertyId::MarginBottom, 
                                        PropertyValue(8));
        
        auto checkbox = storage.createComponent(ComponentType::Checkbox, todo);
        storage.getComponent(checkbox)->properties.setProperty(
            PropertyId::Width, PropertyValue(20));
        
        auto textLabel = storage.createComponent(ComponentType::Label, todo);
        storage.getComponent(textLabel)->properties.setProperty(
            PropertyId::FlexGrow, PropertyValue(1));
        storage.getComponent(textLabel)->properties.setProperty(
            PropertyId::Text, PropertyValue(text));
        
        auto deleteBtn = storage.createComponent(ComponentType::Button, todo);
        storage.getComponent(deleteBtn)->properties.setProperty(
            PropertyId::Width, PropertyValue(60));
        storage.getComponent(deleteBtn)->properties.setProperty(
            PropertyId::Text, PropertyValue("删除"));
        
        todoItems.push_back(todo);
        todoTexts.push_back(text);
        
        engine.markDirty(todoList);
        engine.relayoutIfNeeded();
        dispatcher.onLayoutComplete();
    }
    
    void deleteTodo(size_t index) {
        if (index >= todoItems.size()) return;
        
        storage.destroyComponent(todoItems[index]);
        todoItems.erase(todoItems.begin() + index);
        todoTexts.erase(todoTexts.begin() + index);
        
        engine.markDirty(todoList);
        engine.relayoutIfNeeded();
    }
};
```

        updateLabel();
        
        // 按钮容器
        auto buttonContainer = storage.createComponent(
            ComponentType::Container, container);
        auto* btnContainerEntry = storage.getComponent(buttonContainer);
        btnContainerEntry->properties.setProperty(PropertyId::FlexDirection, 
                                               PropertyValue(FlexDirection::Row));
        btnContainerEntry->properties.setProperty(PropertyId::MarginTop, 
                                                PropertyValue(20));
        
        // 减少按钮
        auto decrementBtn = storage.createComponent(
            ComponentType::Button, buttonContainer);
        auto* decEntry = storage.getComponent(decrementBtn);
        decEntry->properties.setProperty(PropertyId::Width, PropertyValue(60));
        decEntry->properties.setProperty(PropertyId::Height, PropertyValue(40));
        decEntry->properties.setProperty(PropertyId::Text, PropertyValue("-"));
        decEntry->properties.setProperty(PropertyId::MarginRight, PropertyValue(10));
        
        // 增加按钮
        auto incrementBtn = storage.createComponent(
            ComponentType::Button, buttonContainer);
        auto* incEntry = storage.getComponent(incrementBtn);
        incEntry->properties.setProperty(PropertyId::Width, PropertyValue(60));
        incEntry->properties.setProperty(PropertyId::Height, PropertyValue(40));
        incEntry->properties.setProperty(PropertyId::Text, PropertyValue("+"));
        
        engine.forceRelayout();
    }
    
    void updateLabel() {
        std::string text = "Count: " + std::to_string(count);
        storage.getComponent(countLabel)->properties.setProperty(
            PropertyId::Text, PropertyValue(text));
        
        auto* labelEntry = storage.getComponent(countLabel);
        labelEntry->properties.setProperty(PropertyId::FontSize, PropertyValue(48));
    }
    
    void increment() {
        count++;
        updateLabel();
        engine.markDirty(countLabel);
        engine.relayoutIfNeeded();
    }
    
    void decrement() {
        count--;
        updateLabel();
        engine.markDirty(countLabel);
        engine.relayoutIfNeeded();
    }
};

#include "aether/AetherApplication.h"

void createForm(ComponentStorage& storage, LayoutEngine& engine) {

std::tuple<ComponentHandle, ComponentHandle> createFormField(
    ComponentStorage& storage, 
    ComponentHandle parent, 
    const std::string& labelText) {
    
    auto row = storage.createComponent(ComponentType::Container, parent);
    auto* rowEntry = storage.getComponent(row);
    rowEntry->properties.setProperty(PropertyId::FlexDirection, 
                                   PropertyValue(FlexDirection::Row));
    rowEntry->properties.setProperty(PropertyId::MarginBottom, 
                                   PropertyValue(16));
    
    auto label = storage.createComponent(ComponentType::Label, row);
    storage.getComponent(label)->properties.setProperty(
        PropertyId::Width, PropertyValue(60));
    storage.getComponent(label)->properties.setProperty(
        PropertyId::Text, PropertyValue(labelText));
    
    auto input = storage.createComponent(ComponentType::TextField, row);
    storage.getComponent(input)->properties.setProperty(
        PropertyId::FlexGrow, PropertyValue(1));
    
    return std::make_tuple(label, input);
}
